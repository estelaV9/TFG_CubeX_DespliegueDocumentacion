static CubeType loadFromPreferences(SharedPreferences prefs) {
  return CubeType(
    idCube: prefs.getInt("idCube") ?? -1,
    idUser: prefs.getInt("idUser") ?? -1,
    cubeName: prefs.getString("cubeName") ?? '',
  );
}
